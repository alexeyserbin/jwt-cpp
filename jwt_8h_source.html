390  inline std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw,
391  std::error_code& ec) {
392  ec.clear();
393 #if OPENSSL_VERSION_NUMBER <= 0x10100003L
394  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
395  BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
396 #else
397  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
398  BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
399 #endif
400  std::unique_ptr<BIO, decltype(&BIO_free_all)> keybio(BIO_new(BIO_s_mem()), BIO_free_all);
401  if (!certbio || !keybio) {
402  ec = error::rsa_error::create_mem_bio_failed;
403  return {};
404  }
405 
406  std::unique_ptr<X509, decltype(&X509_free)> cert(
407  PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
408  if (!cert) {
409  ec = error::rsa_error::cert_load_failed;
410  return {};
411  }
412  std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
413  if (!key) {
414  ec = error::rsa_error::get_key_failed;
415  return {};
416  }
417  if (PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
418  ec = error::rsa_error::write_key_failed;
419  return {};
420  }
421  char* ptr = nullptr;
422  auto len = BIO_get_mem_data(keybio.get(), &ptr);
423  if (len <= 0 || ptr == nullptr) {
424  ec = error::rsa_error::convert_to_pem_failed;
425  return {};
426  }
427  return {ptr, static_cast<size_t>(len)};
428  }

1439  std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_destroy)> md_ctx(EVP_MD_CTX_create(),
1440  &EVP_MD_CTX_destroy);

1445  ec = error::signature_verification_error::create_context_failed;
1446  return;
1447  }

1449  if (EVP_DigestVerifyInit(md_ctx.get(), &ctx, md(), nullptr, pkey.get()) != 1) {
1450  ec = error::signature_verification_error::verifyinit_failed;
1451  return;
1452  }
1453  if (EVP_PKEY_CTX_set_rsa_padding(ctx, RSA_PKCS1_PSS_PADDING) <= 0) {

1456  }

1458 // sets the salt length to the hash length. Unlike OpenSSL which exposes this functionality.
1459 #ifndef LIBWOLFSSL_VERSION_HEX
1460  if (EVP_PKEY_CTX_set_rsa_pss_saltlen(ctx, -1) <= 0) {
1461  ec = error::signature_verification_error::set_rsa_pss_saltlen_failed;
1462  return;
1463  }
1464 #endif

1466  ec = error::signature_verification_error::verifyupdate_failed;
1467  return;
1468  }
1469 
1470  if (EVP_DigestVerifyFinal(md_ctx.get(), (unsigned char*)signature.data(), signature.size()) <= 0) {
1471  ec = error::signature_verification_error::verifyfinal_failed;
1472  return;
1473  }
1474  }
1479  std::string name() const { return alg_name; }
1480

1483  std::shared_ptr<EVP_PKEY> pkey;
1485  const EVP_MD* (*md)();

1488  };
